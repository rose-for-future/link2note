# 信息处理 Skill Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 一个 Claude Code Skill：给一个链接（抖音/小红书/播客/B站/微博/知乎/公众号/GitHub），自动识别类型→抓取→转写→Claude 总结→输出「文字稿.md + 总结.md」到指定 Obsidian 路径。

**Architecture:** 纯规则 `router` 把 URL 分到平台处理器；每个 `fetcher` 产出统一的 `ContentResult` 字典；`render` 把 `ContentResult` 写成「文字稿.md」并按开关归档素材；`总结.md` 由 Claude（调用方）读文字稿后写入。各平台的脏活（抖音去水印、小红书反爬、SenseVoice 转写）从 chubbyskills（MIT）搬入 `vendor/` 并用薄适配器接到 `ContentResult` 契约上。

**Tech Stack:** Python 3.10+；标准库（urllib/json/re）做抓取；`funasr`+`torch`（SenseVoice 视频转写）；`faster-whisper`（播客）；`ffmpeg`、`curl`、`yt-dlp`（B站字幕）系统依赖；`gh` CLI（GitHub）。

## Global Constraints

- **形态**：Claude Code Skill。所有真实逻辑放 `scripts/`，`SKILL.md` 只编排「何时调哪个脚本」。
- **全本地零付费**：转写用本地开源模型，不调付费云 ASR。分析由 Claude 完成，脚本不内置任何 LLM API 调用。
- **路径不写死**：成稿目录、素材目录、cookie 来源等全部从 `config.json` 读，使用者用前自配。
- **默认成稿目录**：`<你的Obsidian库>/📚知识库/信息处理`
- **默认素材目录**：`<你的Obsidian库>/⚙系统/素材/信息收集原文件`
- **素材策略**：原文链接必留（文字稿顶部）；图片默认存；音频/视频默认**不存**，靠 `keep_audio`/`keep_video` 开关单开。
- **GitHub**：只取 README + 仓库简介，**不取 star/fork/watch 等噪音**。
- **vendor 合规**：搬入 chubbyskills 的文件保留其 MIT LICENSE 与出处注释，集中放 `scripts/vendor/`。
- **文件名清洗**：文件夹名 = 平台原标题经 `clean_title` 清洗（去 `/ \ : * ? " < > |`、折叠空白、截断 80 字符）。
- **测试运行器**：`pytest`。纯逻辑任务必须 TDD；网络/模型任务用冒烟脚本人工验证（标记 `@pytest.mark.smoke`，默认 `-m "not smoke"` 跳过）。

## ContentResult 契约（贯穿全程）

所有 fetcher 产出、render 消费的统一字典（在 Task 3 用 `make_result` 落地）：

```python
  "url": str,          # 原文链接
```

```python
# scripts/config.py
import json, os, copy

DEFAULTS = {
    "notes_dir": "<你的Obsidian库>/📚知识库/信息处理",
    "assets_dir": "<你的Obsidian库>/⚙系统/素材/信息收集原文件",
    "keep_audio": False,
    "keep_video": False,
    "save_images": True,
    "transcribe_backend": "sensevoice",   # sensevoice|faster-whisper|mlx-whisper
    "whisper_model": "small",
    "xhs_cookie": "",                      # 小红书登录态，从浏览器复制；也可用环境变量 XHS_COOKIE
}

_SEARCH = [
    "./config.json",
    os.path.expanduser("~/.claude/skills/信息处理/config.json"),
]

def load_config(path=None):
    cfg = copy.deepcopy(DEFAULTS)
    candidates = [path] if path else _SEARCH
    for c in candidates:
        if c and os.path.isfile(c):
            with open(c, encoding="utf-8") as f:
                user = json.load(f)
            cfg.update({k: v for k, v in user.items() if v is not None})
            break
    if not cfg["xhs_cookie"]:
        cfg["xhs_cookie"] = os.environ.get("XHS_COOKIE", "")
    return cfg
```

```python
# scripts/router.py
import re

# (正则, 平台, 类型) —— 顺序匹配，先到先得
_RULES = [
    (r"(v\.douyin\.com|douyin\.com/video)", "douyin", "video"),
    (r"(xiaohongshu\.com|xhslink\.com)", "xiaohongshu", "image_text"),
    (r"(xiaoyuzhoufm\.com|ximalaya\.com|/rss|\.xml$|\.mp3$)", "podcast", "audio"),
    (r"(bilibili\.com|b23\.tv)", "bilibili", "video"),
    (r"(weibo\.com|weibo\.cn)", "weibo", "video"),
    (r"(zhihu\.com)", "zhihu", "article"),
    (r"(mp\.weixin\.qq\.com)", "wechat", "article"),
    (r"(github\.com)", "github", "repo"),
]

PLATFORM_TYPE = {p: t for _, p, t in _RULES}

def classify(url: str) -> dict:
    for pat, platform, ctype in _RULES:
        if re.search(pat, url, re.IGNORECASE):
            return {"platform": platform, "type": ctype}
    raise ValueError(f"无法识别的链接：{url}")
```

```python
# scripts/models.py
import re

def clean_title(name: str) -> str:
    s = re.sub(r'[/\\:*?"<>|]', "", name or "")
    s = re.sub(r"\s+", " ", s).strip()
    s = s[:80]
    return s or "未命名"

def make_result(platform, ctype, url, title, text, *,
                author="", images=None, media=None, extra=None) -> dict:
    return {
        "platform": platform, "type": ctype, "url": url,
        "title": title, "author": author, "text": text,
        "images": images or [], "media": media or {}, "extra": extra or {},
    }
```

```python
# scripts/render.py
import os, shutil, datetime
from scripts.models import clean_title

def _now():
    return datetime.datetime.now().strftime("%Y-%m-%d %H:%M")

def render(result: dict, cfg: dict) -> dict:
    folder = clean_title(result["title"])
    note_dir = os.path.join(cfg["notes_dir"], folder)
    os.makedirs(note_dir, exist_ok=True)

    # 决定是否需要素材目录
    keep_imgs = cfg.get("save_images", True) and result["images"]
    keep_audio = cfg.get("keep_audio") and result["media"].get("audio")
    keep_video = cfg.get("keep_video") and result["media"].get("video")
    assets_dir = None
    if keep_imgs or keep_audio or keep_video:
        assets_dir = os.path.join(cfg["assets_dir"], folder)
        os.makedirs(assets_dir, exist_ok=True)

    # 复制素材
    img_md = []
    if keep_imgs:
        for im in result["images"]:
            if im.get("path") and os.path.exists(im["path"]):
                dst = os.path.join(assets_dir, os.path.basename(im["path"]))
                shutil.copy2(im["path"], dst)
                img_md.append(f"![]({dst})")
            elif im.get("url"):
                img_md.append(f"![]({im['url']})")
    for kind in ("audio", "video"):
        on = cfg.get(f"keep_{kind}")
        src = result["media"].get(kind)
        if on and src and os.path.exists(src):
            shutil.copy2(src, os.path.join(assets_dir, os.path.basename(src)))

    # 文字稿
    head = [
        f"# {result['title']}", "",
        f"- 原文链接：{result['url']}",
        f"- 平台：{result['platform']}",
        f"- 作者：{result['author']}",
        f"- 采集时间：{_now()}",
        "", "---", "",
    ]
    body = [result["text"]]
    if img_md:
        body += ["", "## 图片", ""] + img_md
    transcript_path = os.path.join(note_dir, "文字稿.md")
    with open(transcript_path, "w", encoding="utf-8") as f:
        f.write("\n".join(head + body) + "\n")

    return {
        "note_dir": note_dir,
        "transcript_path": transcript_path,
        "summary_path": os.path.join(note_dir, "总结.md"),
        "assets_dir": assets_dir,
    }
```

```python
# scripts/transcribe.py
"""统一转写入口：返回纯文字稿，不写文件。"""

_SENSEVOICE = None

def _sensevoice(audio_path: str) -> str:
    global _SENSEVOICE
    from funasr import AutoModel
    from funasr.utils.postprocess_utils import rich_transcription_postprocess
    if _SENSEVOICE is None:
        _SENSEVOICE = AutoModel(
            model="iic/SenseVoiceSmall", trust_remote_code=True,
            vad_model="fsmn-vad", vad_kwargs={"max_single_segment_time": 30000},
            device="cpu",
        )
    res = _SENSEVOICE.generate(input=audio_path, language="zh",
                               use_itn=True, batch_size_s=60)
    return rich_transcription_postprocess(res[0]["text"])

def _faster_whisper(audio_path: str, model_size: str) -> str:
    from faster_whisper import WhisperModel
    model = WhisperModel(model_size, device="cpu", compute_type="int8")
    segments, _ = model.transcribe(audio_path, language="zh")
    return "".join(seg.text for seg in segments).strip()

def transcribe_audio(audio_path: str, cfg: dict) -> str:
    backend = cfg.get("transcribe_backend", "sensevoice")
    if backend == "faster-whisper":
        return _faster_whisper(audio_path, cfg.get("whisper_model", "small"))
    return _sensevoice(audio_path)
```

```python
def fetch(url: str, cfg: dict) -> dict:
```

```python
    state = V.extract_initial_state(html)
```

```python
# scripts/fetchers/github.py
import re, json, subprocess
from scripts.models import make_result
from scripts.fetchers import REGISTRY

def parse_repo_url(url: str):
    m = re.search(r"github\.com/([^/]+)/([^/#?]+)", url)
    if not m:
        raise ValueError(f"非法 GitHub 链接：{url}")
    return m.group(1), m.group(2)

def _build_text(desc: str, readme: str, stars=None) -> str:
    # 故意忽略 stars 等噪音
    parts = []
    if desc:
        parts.append(f"**仓库简介**：{desc}")
    if readme:
        parts.append("## README\n\n" + readme)
    return "\n\n".join(parts)

def _gh_json(owner, repo):
    out = subprocess.run(["gh", "api", f"repos/{owner}/{repo}"],
                         capture_output=True, text=True, check=True).stdout
    return json.loads(out)

def _gh_readme(owner, repo):
    r = subprocess.run(["gh", "api", f"repos/{owner}/{repo}/readme",
                        "-H", "Accept: application/vnd.github.raw"],
                       capture_output=True, text=True)
    return r.stdout if r.returncode == 0 else ""

def fetch(url: str, cfg: dict) -> dict:
    owner, repo = parse_repo_url(url)
    meta = _gh_json(owner, repo)
    readme = _gh_readme(owner, repo)
    text = _build_text(meta.get("description", ""), readme)
    return make_result("github", "repo", url, f"{owner}/{repo}", text,
                       author=owner, extra={"language": meta.get("language")})

REGISTRY["github"] = fetch
```
